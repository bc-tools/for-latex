\begin{bdoctopic}[Contenus mis en avant]
    la sémantique devrait être fixée maintenant via tous les environnements suivants.

    \begin{itemize}
        \item \bdocenv{bdocexa} est inchangé.

        \item \bdocenv{bdocrem} est inchangé.

        \item \bdocenv{bdoctip} est ajouté pour indiquer des astuces.

        \item \bdocenv{bdocnote} remplace l'obsolète \bdocenv{bdocinfo}.

        \item \bdocenv{bdocimportant} est ajouté pour une remarque importante, mais non vitale.

        \item \bdocenv{bdoccaution} est ajouté pour les points délicats.

        \item \bdocenv{bdocwarn} remplace l'obsolète \bdocenv{bdocwarning}.
    \end{itemize}
\end{bdoctopic}
