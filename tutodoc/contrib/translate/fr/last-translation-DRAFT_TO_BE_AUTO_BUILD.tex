\documentclass[twoside]{tutodoc}

\usepackage[french]{babel}

\NewDocumentCommand{\what}{m}{%
	\fbox{\texttt{#1 = ...}}\smallskip%
}


\NewDocumentCommand{\isjust}{m m}{%
	\fbox{\texttt{#1}} = #2
}


\NewDocumentCommand{\onearg}{m m}{
	\begin{enumerate}
		\item \texttt{\#1 = #2}\smallskip

		\item #1{#2}
	\end{enumerate}
}


\NewDocumentCommand{\twoargs}{m m m}{
	\begin{enumerate}
		\item \texttt{\#1 = #2}\smallskip

		\item \texttt{\#2 = #3}\smallskip

		\item #1{#2}{#3}
	\end{enumerate}
}


\NewDocumentCommand{\threeargs}{m m m m}{
	\begin{enumerate}
		\item \texttt{\#1 = #2}\smallskip

		\item \texttt{\#2 = #3}\smallskip

		\item \texttt{\#3 = #4}\smallskip

		\item #1{#2}{#3}{#4}
	\end{enumerate}
}
